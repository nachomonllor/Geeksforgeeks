class Solution {
  public:
    vector<int> increment(vector<int> arr ,int N) {
    
        vector<int> res;
        int ul = arr[N - 1] + 1;
        res.push_back(ul % 10);
        int mellevo = 0;
    
        if (ul > 9) mellevo = 1;
    
        for (int i = N - 2; i >= 0; i--) {
    
            res.push_back((arr[i] + mellevo) % 10);
            if (arr[i] + mellevo > 9) {
                mellevo = 1;
            }
            else {
                mellevo = 0;
            }
    
        }
        if (mellevo > 0) {
            res.push_back(mellevo);
        }
    
        int i  = 0, j = res.size() - 1;
    
        while (i < j) {
            int temp = res[i];
            res[i] = res[j];
            res[j] = temp;
            i++;
            j--;
        }
    
        return res;
        
        
    }
};
