using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace ConsoleApplication51
{
    class Program
    {


        static int MaxZeros(int[] arr)
        {
            int maxZeros = 0;
            int candidate = -1;
            for (int i = 0; i < arr.Length; i++)
            {
                int contZeros = 0;
                int aux = arr[i];
                while (aux > 0)
                {
                    if (aux % 10 == 0)
                    {
                        contZeros++;
                    }

                    aux /= 10;
                }

                if (contZeros > 0)
                {
                    if (contZeros > maxZeros)
                    {
                        maxZeros = contZeros;
                        candidate = arr[i];
                    }
                    else if (contZeros == maxZeros)
                    {
                        if (arr[i] > candidate)
                        {
                            candidate = arr[i];
                        }
                    }
                }
            }
            return candidate;

        }
        


        static void Main(string[] args)
        {

            int t = int.Parse(Console.ReadLine().Trim());

            while (t-- > 0)
            {
                int n = int.Parse(Console.ReadLine().Trim());

                int[] arr = Array.ConvertAll(Console.ReadLine().Trim().Split(' '), e => int.Parse(e));


                Console.WriteLine(MaxZeros(arr));

            }

            




            Console.ReadLine();

        }
    }
}
