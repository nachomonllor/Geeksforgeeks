class Solution{   
public:
    void segregate0and1(int arr[], int n) {
        // code here
        
            int ceros = 0;
            for (int i = 0; i < n; i++) {
                if (arr[i] == 0) {
                    ceros++;
                }
            }
        
            for (int i = 0; i < n; i++) {
                if (ceros > 0) {
                    arr[i] = 0;
                }
                else {
                    arr[i] = 1;
                }
                ceros--;
            }
        
    }
};
