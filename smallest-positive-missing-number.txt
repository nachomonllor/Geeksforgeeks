#include <iostream>
#include <stdio.h>
#include <map>

using namespace std;

int SmallestPositiveMissingNumber(int arr[], int n)
{

    //int[] hash = new int[1000001];
    std::map<int,int> hash;

    for (int i = 0; i < n; i++)
    {
        if (arr[i] > 0)
        {
            hash[arr[i]]++;
        }
    }

    for (int i = 1; i <= 1000000; i++)
    {
        if (hash[i] == 0)
        {
            return i;
        }
    }
    return -1;
}


int main() {
	
	int t;
	cin >> t;
	
	while( t-- ) {
		
		int n;
		cin >> n;
		
		int arr[n];
		for(int i =0; i<n; i++) {
			int elem;
			cin >> elem;
			arr[i] = elem;
		}
		
	    cout <<	SmallestPositiveMissingNumber(arr, n) << endl;
		
		
	}
	
	
	
	return 0;
}
