def remAnagram(str1,str2):
    dic = [0] * 26 

    for item in str1:
        dic[ord(item) - ord('a')] += 1

    for item in str2:
        dic[ord(item) - ord('a')] -= 1

    cont = 0

    for item in dic:
        cont += abs(item ) 
    return cont 

S1 = 'cddgk'
S2 = 'gcd'

print(remAnagram(S1, S2))
