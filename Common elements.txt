class Solution:
    def cargarDiccio(self, A):
        dic = {}
        for item in A:
            if item in dic:
                dic[item] += 1
            else:
                dic[item] = 1
        return dic


    def commonElements (self, A, B, C, n1, n2, n3):
    
        dic_a = self.cargarDiccio(A)
        dic_b = self.cargarDiccio(B)
        dic_c = self.cargarDiccio(C)
    
        res = []
    
        for item in dic_a:
            if item in dic_b and item in dic_c:
                res.append(item)
    
        return sorted(res)
