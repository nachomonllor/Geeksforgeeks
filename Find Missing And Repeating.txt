class Solution:
    def findTwoElement( self,arr, n): 
        # code here
        dic = {}

        for item in arr:
            if item in dic:
                dic[item] += 1
            else:
                dic[item] = 1
    
        res = []
    
        for k in dic:
            if dic[k] == 2:
                res.append(k)
                break
    
        flag = False
        for i in range(1, max(arr)):
            if i not in dic:
                flag = True
                res.append(i)
                break
        if not flag:
            res.append(max(arr) + 1)
    
        return res
