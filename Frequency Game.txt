#include <iostream>
#include <map>

using namespace std;

int LargButMinFreq(int arr[], int n) {
    // code here

    map<int, int> m;

    for (int i = 0; i < n; i++) {
        m[arr[i]]++;
    }

    int min_frec = n;

    for (std::map<int, int>::iterator it = m.begin(); it != m.end(); it++) {
        min_frec = std::min(min_frec, it->second);
    }

    int max_min_frec = -1;
    for (int i = 0; i < n; i++) {
        if (m[arr[i]] == min_frec) {
            max_min_frec = std::max(max_min_frec, arr[i]);
        }
    }

    return max_min_frec;
}


int main()
{
    int arr[] = { 2, 2 ,5, 50, 1 };
    cout << LargButMinFreq(arr, 5) << endl;
    return 0;
}
