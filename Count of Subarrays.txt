#include <iostream>
#include <stdio.h>
#include <vector>
#include <map>
#include <string.h>
#include <math.h>

#define ll long long int

using namespace std;

int fact(int n) {
  int f=1;
  for(int i=2; i<=n; i++) {
      f*=i;   
   }   
   return f;
}

int C(int n, int m){
   return fact(m)/(fact(n) * fact(m-n) );  
}

int main() {
  
  int t;
  scanf("%d", &t);
  
  while(t--) {
    
    int N, K;
    scanf("%d %d", &N, &K);
    
    int arr[N];
    for(int i =0; i<N; i++) {
       scanf("%d", &arr[i]);   
    }
    
    int cont =0;
    for(int i =0; i < N; i++) {
      if(arr[i] > K) 
            cont++;       
    }
    
    printf("%d", (cont * N)   - C(2,N));  
   
  }
  
    
}
